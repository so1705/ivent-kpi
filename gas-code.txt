function getStableCurrentCell_(sheet) {
  SpreadsheetApp.flush();
  Utilities.sleep(120);
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const selection = ss.getSelection();
  let cell = null;
  try {
    if (selection && selection.getCurrentCell && selection.getCurrentCell()) {
      cell = selection.getCurrentCell();
    }
  } catch (e) {}
  if (!cell) {
    try { cell = ss.getCurrentCell(); } catch (e) {}
  }
  if (!cell && sheet.getActiveCell) {
    try { cell = sheet.getActiveCell(); } catch (e) {}
  }
  if (!cell && selection && selection.getActiveRange && selection.getActiveRange()) {
    try { cell = selection.getActiveRange().getCell(1, 1); } catch (e) {}
  }
  if (!cell) return null;
  if (cell.getSheet().getName() !== sheet.getName()) return null;
  return cell;
}


function getStableCurrentRow_(sheet) {
  const cell = getStableCurrentCell_(sheet);
  return cell ? cell.getRow() : -1;
}

const SELECTION_CTX_KEY = 'MASTER_COMPANY_SELECTION_CTX';

function onSelectionChange(e) {
  try {
    if (!e || !e.range) return;
    const range = e.range;
    const sheet = range.getSheet();
    if (!sheet || sheet.getName() !== CONFIG.SS_MASTER) return;
    if (range.getColumn() !== CONFIG.M_COMPANY) return;
    if (range.getNumRows() !== 1 || range.getNumColumns() !== 1) return;

    ensureMasterSheetStructure_();
    const row = range.getRow();
    if (row <= 1) return;

    const company = String(range.getDisplayValue() || range.getValue() || '').trim();
    if (!company) return;

    const idCell = sheet.getRange(row, CONFIG.M_COMPANY_ID);
    let companyId = String(idCell.getValue() || '').trim();
    if (!companyId) {
      companyId = generateUniqueId_('CMP');
      idCell.setValue(companyId);
    }

    const payload = {
      sheetName: sheet.getName(),
      row: row,
      column: range.getColumn(),
      company: company,
      companyId: companyId,
      at: Date.now()
    };
    PropertiesService.getUserProperties().setProperty(SELECTION_CTX_KEY, JSON.stringify(payload));
  } catch (err) {
    console.error('onSelectionChange error: ' + err.message);
  }
}

function getStoredMasterSelectionContext_() {
  try {
    const raw = PropertiesService.getUserProperties().getProperty(SELECTION_CTX_KEY);
    if (!raw) return null;
    const parsed = JSON.parse(raw);
    if (!parsed || parsed.sheetName !== CONFIG.SS_MASTER) return null;
    if (!parsed.row || !parsed.company) return null;
    if (Date.now() - Number(parsed.at || 0) > 1000 * 60 * 30) return null;
    return parsed;
  } catch (err) {
    return null;
  }
}

function clearStoredMasterSelectionContext_() {
  try { PropertiesService.getUserProperties().deleteProperty(SELECTION_CTX_KEY); } catch (err) {}
}

function ensureMasterIdColumn_(sheet) {
  // 旧呼び出し互換。必要列を用意し、既存IDを補完する。
  ensureMasterSheetStructure_();

    if (sheet.getMaxColumns() >= CONFIG.M_COMPANY_ID) sheet.hideColumns(CONFIG.M_COMPANY_ID);

    if (master.getMaxColumns() >= CONFIG.M_COMPANY_ID) master.hideColumns(CONFIG.M_COMPANY_ID);
    backfillMasterCompanyIds_(master);
  }
}

function getSelectedMasterCompanyContext_(sheet) {
  ensureMasterIdColumn_(sheet);

  const stored = getStoredMasterSelectionContext_();
  let selectedName = '';
  let currentRow = -1;
  let hintedId = '';

  if (stored) {
    selectedName = String(stored.company || '').trim();
    currentRow = Number(stored.row || -1);
    hintedId = String(stored.companyId || '').trim();
  } else {
    const cell = getStableCurrentCell_(sheet);
    if (!cell) return { ok: false, message: '会社名セル（E列）をクリックしてから実行してください。' };
    if (cell.getColumn() !== CONFIG.M_COMPANY) {
      return { ok: false, message: '企業マスターでは「会社名」セル（E列）をクリックしてから実行してください。' };
    }
    selectedName = String(cell.getDisplayValue() || cell.getValue() || '').trim();
    currentRow = cell.getRow();
    hintedId = String(sheet.getRange(currentRow, CONFIG.M_COMPANY_ID).getValue() || '').trim();
  }

  if (!selectedName) return { ok: false, message: '会社名セルが空です。会社名が入っているセルを選んでください。' };

  const lastRow = Math.max(sheet.getLastRow(), 2);
  if (lastRow < 2) return { ok: false, message: '企業マスターにデータがありません。' };

  const companyVals = sheet.getRange(2, CONFIG.M_COMPANY, lastRow - 1, 1).getDisplayValues();
  const idVals = sheet.getRange(2, CONFIG.M_COMPANY_ID, lastRow - 1, 1).getValues();
  const candidates = [];
  for (let i = 0; i < companyVals.length; i++) {
    const name = String(companyVals[i][0] || '').trim();
    if (name === selectedName) {
      let id = String(idVals[i][0] || '').trim();
      const row = i + 2;
      if (!id) {
        id = generateUniqueId_('CMP');
        sheet.getRange(row, CONFIG.M_COMPANY_ID).setValue(id);
      }
      candidates.push({ row: row, company: name, companyId: id });
    }
  }

  if (candidates.length === 0) {
    return { ok: false, message: `会社名「${selectedName}」に一致する行が見つかりません。` };
  }

  if (hintedId) {
    const exactById = candidates.find(c => c.companyId === hintedId);
    if (exactById) return { ok: true, row: exactById.row, company: exactById.company, companyId: exactById.companyId };
  }

  const exactByRow = candidates.find(c => c.row === currentRow);
  if (exactByRow) return { ok: true, row: exactByRow.row, company: exactByRow.company, companyId: exactByRow.companyId };

  if (candidates.length === 1) return { ok: true, row: candidates[0].row, company: candidates[0].company, companyId: candidates[0].companyId };

  return {
    ok: false,
    message: `会社名「${selectedName}」が複数行あります。会社名セルを押し直してから実行するか、重複行を整理してください。候補行: ${candidates.map(c => c.row).join(', ')}`
  };
}



function getMasterCompanyCandidatesForDialog_(sheet) {
  ensureMasterIdColumn_(sheet);
  const lastRow = Math.max(sheet.getLastRow(), 2);
  if (lastRow < 2) return [];

  const numRows = lastRow - 1;
  const companies = sheet.getRange(2, CONFIG.M_COMPANY, numRows, 1).getDisplayValues();
  const ids = sheet.getRange(2, CONFIG.M_COMPANY_ID, numRows, 1).getValues();
  const tels = sheet.getRange(2, CONFIG.M_TEL, numRows, 1).getDisplayValues();
  const staffs = sheet.getRange(2, CONFIG.M_STAFF, numRows, 1).getDisplayValues();
  const emails = sheet.getRange(2, CONFIG.M_EMAIL, numRows, 1).getDisplayValues();
  const appointers = sheet.getRange(2, CONFIG.M_APPOINTER, numRows, 1).getDisplayValues();
  const meetingReps = sheet.getRange(2, CONFIG.M_MEETING_REP, numRows, 1).getDisplayValues();

  const candidates = [];
  for (let i = 0; i < numRows; i++) {
    const row = i + 2;
    const company = String(companies[i][0] || '').trim();
    if (!company) continue;
    if (sheet.isRowHiddenByFilter && sheet.isRowHiddenByFilter(row)) continue;
    if (sheet.isRowHiddenByUser && sheet.isRowHiddenByUser(row)) continue;

    let companyId = String(ids[i][0] || '').trim();
    if (!companyId) {
      companyId = generateUniqueId_('CMP');
      sheet.getRange(row, CONFIG.M_COMPANY_ID).setValue(companyId);
    }

    candidates.push({
      row: row,
      companyId: companyId,
      company: company,
      tel: String(tels[i][0] || '').trim(),
      staff: String(staffs[i][0] || '').trim(),
      email: String(emails[i][0] || '').trim(),
      appointer: String(appointers[i][0] || '').trim(),
      meetingRep: String(meetingReps[i][0] || '').trim()
    });
  }
  return candidates;
}

function getPreferredMasterCandidateId_(sheet, candidates) {
  if (!candidates || !candidates.length) return '';
  const current = getStableCurrentCell_(sheet);
  if (current && current.getSheet().getName() === sheet.getName()) {
    const row = current.getRow();
    const hit = candidates.find(c => c.row === row);
    if (hit) return hit.companyId;
  }
  const stored = getStoredMasterSelectionContext_();
  if (stored && stored.companyId) {
    const hit = candidates.find(c => c.companyId === stored.companyId);
    if (hit) return hit.companyId;
  }
  return String(candidates[0].companyId || '');
}

function setupMasterSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(CONFIG.SS_MASTER);
  if (sheet) ss.deleteSheet(sheet);
  sheet = ss.insertSheet(CONFIG.SS_MASTER);

  sheet.getRange(1, 1, 1, MASTER_HEADERS.length).setValues([MASTER_HEADERS]);
  sheet.getRange(1, 1, 1, MASTER_HEADERS.length)
    .setBackground('#1a73e8').setFontColor('#ffffff')
    .setFontWeight('bold').setHorizontalAlignment('center');
  sheet.setFrozenRows(1);

  const widths = [46,110, 120, 100, 100, 220, 130, 120, 220, 220, 120, 80, 110, 300, 280, 220, 110, 120];
  widths.forEach((w, i) => sheet.setColumnWidth(i + 1, w));

  const dataRows = 1000;
  const startRow = 2;
  const endRow = startRow + dataRows - 1;

  sheet.getRange(startRow, CONFIG.M_TARGET, dataRows, 1).insertCheckboxes();
  setDropdown(sheet, CONFIG.M_STATUS, startRow, endRow, STATUS_OPTIONS);
  setDropdown(sheet, CONFIG.M_APPOINTER, startRow, endRow, MEMBER_OPTIONS);
  setDropdown(sheet, CONFIG.M_MEETING_REP, startRow, endRow, [''].concat(MEMBER_OPTIONS));
  setDropdown(sheet, CONFIG.M_INDUSTRY, startRow, endRow, [''].concat(INDUSTRY_OPTIONS));
  sheet.getRange(startRow, CONFIG.M_CHECKBOX, dataRows, 1).insertCheckboxes();

  const rules = [];
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenFormulaSatisfied('=AND($C2<>"", $C2<TODAY())')
    .setBackground('#f8bbd0')
    .setRanges([sheet.getRange(`B${startRow}:B${endRow}`)]).build());
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenFormulaSatisfied('=$C2=TODAY()')
    .setBackground('#ef5350').setFontColor('#ffffff')
    .setRanges([sheet.getRange(`B${startRow}:B${endRow}`)]).build());
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenFormulaSatisfied('=$C2=TODAY()+1')
    .setBackground('#fff59d')
    .setRanges([sheet.getRange(`B${startRow}:B${endRow}`)]).build());
  sheet.setConditionalFormatRules(rules);
  sheet.hideColumns(CONFIG.M_COMPANY_ID);
  backfillMasterCompanyIds_(sheet);
}

function backfillMasterCompanyIds_(sheet) {
  const lastRow = Math.max(sheet.getLastRow(), 2);
  if (lastRow < 2) return;
  const companyVals = sheet.getRange(2, CONFIG.M_COMPANY, lastRow - 1, 1).getValues();
  const idVals = sheet.getRange(2, CONFIG.M_COMPANY_ID, lastRow - 1, 1).getValues();
  let changed = false;
  const seen = {};
  for (let i = 0; i < companyVals.length; i++) {
    const company = String(companyVals[i][0] || '').trim();
    let id = String(idVals[i][0] || '').trim();
    if (!company) continue;
    if (!id || seen[id]) {
      id = generateUniqueId_('CMP');
      idVals[i][0] = id;
      changed = true;
    }
    seen[id] = true;
  }
  if (changed) sheet.getRange(2, CONFIG.M_COMPANY_ID, lastRow - 1, 1).setValues(idVals);
}

function findRowById_(sheet, id, colIndex) {
  const targetId = String(id || '').trim();
  if (!targetId) return 0;
  const lastRow = Math.max(sheet.getLastRow(), 2);
  if (lastRow < 2) return 0;
  const values = sheet.getRange(2, colIndex, lastRow - 1, 1).getValues();
  for (let i = 0; i < values.length; i++) {
    if (String(values[i][0] || '').trim() === targetId) return i + 2;
  }
  return 0;
}

function resolveMasterRow_(sheet, payload) {
  const rowHint = Number(payload.rowHint || 0);
  const companyId = String(payload.companyId || '').trim();
  const companyName = String(payload.companyName || '').trim();
  if (rowHint > 1 && rowHint <= sheet.getMaxRows()) {
    const hintId = String(sheet.getRange(rowHint, CONFIG.M_COMPANY_ID).getValue() || '').trim();
    const hintCompany = String(sheet.getRange(rowHint, CONFIG.M_COMPANY).getValue() || '').trim();
    if ((!companyId || hintId === companyId) && (!companyName || hintCompany === companyName)) return rowHint;
  }
  if (companyId) {
    const byId = findRowById_(sheet, companyId, CONFIG.M_COMPANY_ID);
    if (byId) return byId;
  }
  if (companyName) {
    const lastRow = Math.max(sheet.getLastRow(), 2);
    const companies = sheet.getRange(2, CONFIG.M_COMPANY, lastRow - 1, 1).getValues();
    for (let i = 0; i < companies.length; i++) {
      if (String(companies[i][0] || '').trim() === companyName) return i + 2;
    }
  }
  return 0;
}

function withSubmissionGuard_(guardKey, ttlSeconds, callback) {
  const cache = CacheService.getScriptCache();
  const existing = cache.get(guardKey);
  if (existing) return '反映中、または直前の同じ操作をスキップしました。';
  const lock = LockService.getScriptLock();
  lock.waitLock(15000);
  try {
    if (cache.get(guardKey)) return '反映中、または直前の同じ操作をスキップしました。';
    cache.put(guardKey, '1', ttlSeconds || 12);
    const result = callback();
    return result;
  } catch (err) {
    cache.remove(guardKey);
    throw err;
  } finally {
    lock.releaseLock();
  }
}

function getSettings() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(CONFIG.SS_SETTINGS);
  if (!sheet) throw new Error('設定シートが見つかりません。');

  const ccEmail = sheet.getRange(CONFIG.S_CC_EMAIL_CELL).getValue() || '';
  const infoCalendarId = sheet.getRange(CONFIG.S_INFO_CALENDAR_ID_CELL).getValue() || '';
  const memberRange = sheet.getRange('A6:E20').getValues();
  const members = {};
  memberRange.forEach(row => {
    if (row[0]) {
      members[row[0]] = {
        uni: row[1] || '',
        full: row[2] || row[0],
        kana: row[3] || '',
        email: row[4] || ''
      };
    }
  });
  const urlRange = sheet.getRange('A27:B30').getValues();
  const materialUrls = {};
  urlRange.forEach(row => { if (row[0]) materialUrls[row[0]] = row[1] || ''; });
  const eventStatusOptions = sheet.getRange('A33:A80').getValues().flat().map(v => String(v || '').trim()).filter(Boolean);
  return { ccEmail, infoCalendarId, members, materialUrls, eventStatusOptions: eventStatusOptions.length ? [''].concat(eventStatusOptions) : EVENT_STATUS_OPTIONS };
}

function guessMemberNameByEmail_(email, members) {
  const normalized = String(email || '').trim().toLowerCase();
  for (const name in members) {
    if (String(members[name].email || '').trim().toLowerCase() === normalized) return name;
  }
  return '';
}

function submitTelCallRecord(payload) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(CONFIG.SS_MASTER);
  ensureMasterSheetStructure_();
  ensureActiveSheetStructure_();
  const companyId = String(payload.companyId || '').trim();
  const companyName = String(payload.companyName || '').trim();
  const row = resolveMasterRow_(sheet, payload);
  if (!sheet || !row || row <= 1) throw new Error('企業マスターの対象行が見つかりません。フィルター中でも企業IDで記録できるよう修正済みですが、対象データを確認してください。');

  const result = String(payload.result || '').trim();
  const status = String(payload.status || '').trim();
  const memoInput = String(payload.memoInput || '').trim();
  const nextDateInput = String(payload.nextDateInput || '').trim();
  const appointer = String(payload.appointer || '').trim();
  const meetingRep = String(payload.meetingRep || '').trim();
  const staffName = String(payload.staffName || '').trim();
  const email = String(payload.email || '').trim();
  if (!result || !status) throw new Error('結果またはステータスが空です。');
  if (!appointer) throw new Error('アポ獲得者を選んでください。');

  const guardKey = `TEL:${companyId || companyName}:${status}:${result}:${memoInput}:${nextDateInput}`;
  return withSubmissionGuard_(guardKey, 15, () => {
  let nextDate = null;
  if (nextDateInput) {
    nextDate = parseFlexibleDateTime(nextDateInput);
    if (!nextDate) throw new Error('次回架電予定日の形式が正しくありません。');
  }

  const today = new Date();
  const todayStr = Utilities.formatDate(today, 'Asia/Tokyo', 'M/d');
  const currentCount = Number(sheet.getRange(row, CONFIG.M_CALL_COUNT).getValue()) || 0;

  sheet.getRange(row, CONFIG.M_STATUS).setValue(status);
  sheet.getRange(row, CONFIG.M_APPOINTER).setValue(appointer);
  sheet.getRange(row, CONFIG.M_MEETING_REP).setValue(meetingRep || '');
  if (staffName) sheet.getRange(row, CONFIG.M_STAFF).setValue(staffName);
  if (email) sheet.getRange(row, CONFIG.M_EMAIL).setValue(email);
  if (memoInput) appendMultilineCell_(sheet.getRange(row, CONFIG.M_MEMO), `[${todayStr} ${appointer}/${status}] ${memoInput}`);
  sheet.getRange(row, CONFIG.M_CALL_COUNT).setValue(currentCount + 1);
  sheet.getRange(row, CONFIG.M_LAST_CALL).setValue(today);
  if (nextDate) sheet.getRange(row, CONFIG.M_NEXT_DATE).setValue(nextDate);

  const industry = sheet.getRange(row, CONFIG.M_INDUSTRY).getValue() || '未設定';
  syncToFirestore_(appointer, status, industry);

  const historyCell = sheet.getRange(row, CONFIG.M_HISTORY);
  const existingHistory = historyCell.getValue();
  const numberPrefix = ['①','②','③','④','⑤','⑥','⑦','⑧','⑨','⑩'][currentCount] || `(${currentCount + 1})`;
  const memoSuffix = memoInput ? ` [${memoInput}]` : '';
  const newEntry = `${numberPrefix}${todayStr} ${appointer}:${result}${memoSuffix}`;
  historyCell.setValue(existingHistory ? existingHistory + '\n' + newEntry : newEntry);

  if (memoInput) {
    const talkPointCell = sheet.getRange(row, CONFIG.M_TALK_POINT);
    const existingTalkPoint = talkPointCell.getValue();
    const talkPointEntry = `[${todayStr} ${appointer}/${status}] ${memoInput}`;
    talkPointCell.setValue(existingTalkPoint ? existingTalkPoint + '\n' + talkPointEntry : talkPointEntry);
  }

  let message = `記録完了 (${status} / 累計${currentCount + 1}回目)`;
  if (isTransferStatus_(status)) {
    const activeRow = upsertToActiveFromMaster_(row, status);
    sheet.getRange(row, CONFIG.M_CHECKBOX).setValue(true);
    sheet.getRange(row, CONFIG.M_TRANSFER).setValue(`自動反映済 (アクティブ${activeRow}行目)`);
    message += `\nアクティブ案件へ自動反映しました。`;
  }

  clearTargetCheckbox_(sheet, row, CONFIG.M_TARGET);
  SpreadsheetApp.getActiveSpreadsheet().toast(message, '📞 テレアポ記録', 4);
  return message;
  });
}

function isTransferStatus_(status) {
  return ['資料送付予定A','資料送付予定B','資料送付予定C','アポ確定'].indexOf(String(status || '')) >= 0;
}

function submitAppointmentSetup(payload) {

  ensureActiveSheetStructure_();
  const row = resolveActiveRow_(sheet, payload);
  if (!sheet || row <= 1) throw new Error('アクティブ案件の対象行が見つかりません。');
  const settings = getSettings();
  const targetCalendarId = settings.infoCalendarId || '';
  if (!targetCalendarId) throw new Error('設定シートB3に予定作成先カレンダーIDを入れてください。');
  const calMeta = Calendar.CalendarList.get(targetCalendarId);
  const accessRole = calMeta.accessRole || '';
  if (accessRole !== 'owner' && accessRole !== 'writer') throw new Error(`このアカウント(${MY_EMAIL})には ${targetCalendarId} への編集権限がありません。`);

  const company = sheet.getRange(row, CONFIG.A_COMPANY).getValue();
  const email = String(payload.email || '').trim();
  const staff = String(payload.staff || '').trim();
  const appointer = String(payload.appointer || '').trim();
  const meetingRep = String(payload.meetingRep || '').trim() || appointer;
  const memo = String(payload.memo || '').trim();
  const startDate = parseFlexibleDateTime(payload.meetTime, true);
  if (!company || !email || !staff || !appointer || !startDate) throw new Error('会社名・メール・担当者名・アポ獲得者・商談日時は必須です。');
  const guardKey = `APT:${payload.recordId || company}:${email}:${payload.meetTime}:${appointer}`;
  return withSubmissionGuard_(guardKey, 20, () => {
  const endDate = new Date(startDate.getTime() + 60 * 60 * 1000);

  sheet.getRange(row, CONFIG.A_STAFF).setValue(staff);
  sheet.getRange(row, CONFIG.A_EMAIL).setValue(email);
  sheet.getRange(row, CONFIG.A_APPOINTER).setValue(appointer);
  sheet.getRange(row, CONFIG.A_MEETING_REP).setValue(meetingRep);
  sheet.getRange(row, CONFIG.A_MEET_TIME).setValue(startDate);
  if (memo) appendMultilineCell_(sheet.getRange(row, CONFIG.A_MEET_MEMO), memo);

  const myData = settings.members[appointer] || { full: appointer };
  const repData = settings.members[meetingRep] || { full: meetingRep };
  const appointerLastName = String(myData.full || appointer).split(/[ 　]/)[0];
  const repLastName = String(repData.full || meetingRep).split(/[ 　]/)[0];
  const calendarTitle = `${company} ${staff}様: 新卒採用イベント_お打ち合わせ`;
  const repIntro = (appointer === meetingRep || !meetingRep)
    ? `当日は私${appointerLastName}がお話しさせていただきます。`
    : `当日は弊社の${repLastName}がお話しさせていただきます。`;

    // カレンダー作成のコード付近に挿入
  syncToFirestore_(appointer, 'アポセットアップ', 'アクティブ案件');

  let meetUrl = '';
  let calendarSuccess = false;
  let calendarError = '';
  try {
    const requestId = 'meet-' + Date.now() + '-' + Math.random().toString(36).substring(2, 10);
    const eventResource = {
      summary: calendarTitle,
      description: `会社名: ${company}\n担当者: ${staff}\n商談担当: ${meetingRep}\nメモ:\n${sheet.getRange(row, CONFIG.A_MEET_MEMO).getValue() || ''}`,
      start: { dateTime: startDate.toISOString(), timeZone: 'Asia/Tokyo' },

    };

    if (createdEvent.conferenceData && createdEvent.conferenceData.entryPoints) {
      meetUrl = createdEvent.conferenceData.entryPoints[0].uri;
    } else {
      Utilities.sleep(1500);
      const refetched = Calendar.Events.get(targetCalendarId, createdEvent.id);
      if (refetched.conferenceData && refetched.conferenceData.entryPoints) meetUrl = refetched.conferenceData.entryPoints[0].uri;
    }
    calendarSuccess = true;

    calendarError = err.message;

function onEdit(e) {
  if (!e || !e.range) return;
  const sheet = e.source.getActiveSheet();
  const range = e.range;
  const sheetName = sheet.getName();

  if (sheetName === CONFIG.SS_MASTER && range.getColumn() === CONFIG.M_CHECKBOX && e.value === 'TRUE') {
    const row = range.getRow();
    const statusA = String(sheet.getRange(row, CONFIG.M_STATUS).getValue());
    if (isTransferStatus_(statusA)) {
      try {
        const targetRow = upsertToActiveFromMaster_(row, statusA);
        sheet.getRange(row, CONFIG.M_TRANSFER).setValue(`手動反映済 (アクティブ${targetRow}行目)`);
      } catch(err) {
        sheet.getRange(row, CONFIG.M_TRANSFER).setValue('エラー: ' + err.message);
        console.error('transferToActive error: ' + err.message);
      }
    } else {
      sheet.getRange(row, CONFIG.M_TRANSFER).setValue('対象外: ' + statusA);
    }
  }
}

function upsertToActiveFromMaster_(masterRow, statusA) {
  ensureActiveSheetStructure_();
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sourceSheet = ss.getSheetByName(CONFIG.SS_MASTER);
  const target = ss.getSheetByName(CONFIG.SS_ACTIVE);
  const company = String(sourceSheet.getRange(masterRow, CONFIG.M_COMPANY).getValue() || '').trim();
  const tel = String(sourceSheet.getRange(masterRow, CONFIG.M_TEL).getValue() || '').trim();
  if (!company) throw new Error('会社名が空です。');
  let companyId = String(sourceSheet.getRange(masterRow, CONFIG.M_COMPANY_ID).getValue() || '').trim();
  if (!companyId) {
    companyId = generateUniqueId_('CMP');
    sourceSheet.getRange(masterRow, CONFIG.M_COMPANY_ID).setValue(companyId);
  }

  let targetRow = findRowById_(target, companyId, CONFIG.A_RECORD_ID);
  if (!targetRow) targetRow = findActiveRowByCompanyAndTel_(target, company, tel);
  if (!targetRow) targetRow = getNextBlankActiveRow_(target);

  const staff = sourceSheet.getRange(masterRow, CONFIG.M_STAFF).getValue() || '';
  const email = sourceSheet.getRange(masterRow, CONFIG.M_EMAIL).getValue() || '';
  const appointer = sourceSheet.getRange(masterRow, CONFIG.M_APPOINTER).getValue() || '';
  const meetingRep = sourceSheet.getRange(masterRow, CONFIG.M_MEETING_REP).getValue() || '';
  const memo = sourceSheet.getRange(masterRow, CONFIG.M_MEMO).getValue() || '';
  const talk = sourceSheet.getRange(masterRow, CONFIG.M_TALK_POINT).getValue() || '';
  const hist = sourceSheet.getRange(masterRow, CONFIG.M_HISTORY).getValue() || '';
  const nextDate = sourceSheet.getRange(masterRow, CONFIG.M_NEXT_DATE).getValue() || '';

  target.getRange(targetRow, CONFIG.A_RECORD_ID).setValue(companyId);
  target.getRange(targetRow, CONFIG.A_COMPANY).setValue(company);
  target.getRange(targetRow, CONFIG.A_TEL).setValue(tel);
  if (staff) target.getRange(targetRow, CONFIG.A_STAFF).setValue(staff);
  if (email) target.getRange(targetRow, CONFIG.A_EMAIL).setValue(email);
  if (appointer) target.getRange(targetRow, CONFIG.A_APPOINTER).setValue(appointer);
  if (meetingRep) target.getRange(targetRow, CONFIG.A_MEETING_REP).setValue(meetingRep);
  if (hist) target.getRange(targetRow, CONFIG.A_HISTORY).setValue(hist);
  if (nextDate) target.getRange(targetRow, CONFIG.A_NEXT_DATE).setValue(nextDate);

  const aggregatedMemoParts = [];
  if (memo) aggregatedMemoParts.push(`[最新メモ]\n${memo}`);
  if (talk) aggregatedMemoParts.push(`[次回トーク要点履歴]\n${talk}`);
  const aggregatedMemo = aggregatedMemoParts.join('\n\n');
  if (aggregatedMemo) {
    target.getRange(targetRow, CONFIG.A_MEET_MEMO).setValue(aggregatedMemo);
  }

  let kind = '';
  let docRank = '';
  if (statusA === 'アポ確定') kind = 'アポ確定';
  if (statusA === '資料送付予定A') { kind = '資料送付'; docRank = '資料送付予定A'; }
  if (statusA === '資料送付予定B') { kind = '資料送付'; docRank = '資料送付予定B'; }
  if (statusA === '資料送付予定C') { kind = '資料送付'; docRank = '資料送付予定C'; }
  if (kind) target.getRange(targetRow, CONFIG.A_KIND).setValue(kind);
  if (docRank) target.getRange(targetRow, CONFIG.A_DOC_RANK).setValue(docRank);
  if (!target.getRange(targetRow, CONFIG.A_PRIORITY).getValue()) target.getRange(targetRow, CONFIG.A_PRIORITY).setValue('⭐中');
  target.getRange(targetRow, CONFIG.A_ACTION_COUNT).setValue(Number(target.getRange(targetRow, CONFIG.A_ACTION_COUNT).getValue()) || 0);
  return targetRow;
}

function findActiveRowByCompanyAndTel_(sheet, company, tel) {
  const lastRow = Math.max(sheet.getLastRow(), 2);
  if (lastRow < 2) return 0;
  const data = sheet.getRange(2, CONFIG.A_COMPANY, lastRow - 1, 3).getValues();
  for (let i = 0; i < data.length; i++) {
    const c = String(data[i][0] || '').trim();
    const t = String(data[i][2] || '').trim();
    if (c && c === company && t === tel) return i + 2;
  }
  return 0;
}

function getNextBlankActiveRow_(sheet) {
  const lastRow = Math.max(sheet.getLastRow(), 1);
  for (let r = 2; r <= Math.max(lastRow, 1000); r++) {
    const company = sheet.getRange(r, CONFIG.A_COMPANY).getValue();
    if (!company) return r;
  }
  sheet.insertRowsAfter(sheet.getMaxRows(), 100);
  applyActiveSheetFormatting_(sheet);
  return sheet.getLastRow() - 99;
}

function submitDocFollowupRecord(payload) {
  if (!ensurePermissions()) throw new Error('権限承認が必要です。');
  ensureActiveSheetStructure_();
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CONFIG.SS_ACTIVE);
  ensureActiveSheetStructure_();
  const row = resolveActiveRow_(sheet, payload);
  if (!sheet || row <= 1) throw new Error('アクティブ案件の対象行が見つかりません。');
  const settings = getSettings();
  const targetCalendarId = settings.infoCalendarId || '';
  const company = sheet.getRange(row, CONFIG.A_COMPANY).getValue() || '';
  const rep = sheet.getRange(row, CONFIG.A_MEETING_REP).getValue() || sheet.getRange(row, CONFIG.A_APPOINTER).getValue() || '';
  const result = String(payload.result || '').trim();
  const june = String(payload.june || '');
  const aug = String(payload.aug || '');
  const nov = String(payload.nov || '');
  const nextMonth = String(payload.nextMonth || '');
  const nextMethod = String(payload.nextMethod || '');
  const memo = String(payload.memo || '').trim();
  const nextDate = parseFlexibleDateTime(payload.nextDate);

  if (payload.nextDate && !nextDate) throw new Error('次回後追い日の形式が正しくありません。');
  const guardKey = `DOC:${payload.recordId || company}:${result}:${payload.nextDate}:${memo}`;
  return withSubmissionGuard_(guardKey, 15, () => {

  if (june || payload.june === '') sheet.getRange(row, CONFIG.A_JUNE_STATUS).setValue(june);
  if (aug || payload.aug === '') sheet.getRange(row, CONFIG.A_AUG_STATUS).setValue(aug);
  if (nov || payload.nov === '') sheet.getRange(row, CONFIG.A_NOV_STATUS).setValue(nov);
  if (nextMonth || payload.nextMonth === '') sheet.getRange(row, CONFIG.A_NEXT_MONTH).setValue(nextMonth);
  if (nextDate) {
    sheet.getRange(row, CONFIG.A_NEXT_DATE).setValue(nextDate);
    sheet.getRange(row, CONFIG.A_NEXT_APPROACH_DATE).setValue(nextDate);
  }
  sheet.getRange(row, CONFIG.A_NEXT_KIND).setValue('資料後追い');
  if (nextMethod || payload.nextMethod === '') sheet.getRange(row, CONFIG.A_NEXT_METHOD).setValue(nextMethod);
  if (memo || payload.memo === '') sheet.getRange(row, CONFIG.A_NEXT_CONTENT).setValue(memo);

  const actionCount = Number(sheet.getRange(row, CONFIG.A_ACTION_COUNT).getValue()) || 0;
  const today = new Date();
  const todayStr = Utilities.formatDate(today, 'Asia/Tokyo', 'M/d');
  const prefix = ['①','②','③','④','⑤','⑥','⑦','⑧','⑨','⑩'][actionCount] || `(${actionCount + 1})`;
  const historyEntry = `${prefix}${todayStr} ${rep || '担当未設定'}:資料後追い/${result}${memo ? ' [' + memo + ']' : ''}`;
  appendMultilineCell_(sheet.getRange(row, CONFIG.A_HISTORY), historyEntry);
  if (memo) appendMultilineCell_(sheet.getRange(row, CONFIG.A_MEET_MEMO), `[${todayStr}] 資料後追い: ${memo}`);
  sheet.getRange(row, CONFIG.A_ACTION_COUNT).setValue(actionCount + 1);
  sheet.getRange(row, CONFIG.A_LAST_ACTION).setValue(today);
  sheet.getRange(row, CONFIG.A_LAST_RESULT).setValue(`資料後追い:${result}`);

  syncToFirestore_(rep, `資料後追い:${result}`, '資料後追い');

  if (result === 'アポ確定') {
    sheet.getRange(row, CONFIG.A_KIND).setValue('アポ確定');
  } else {
    sheet.getRange(row, CONFIG.A_KIND).setValue('資料送付');
  }

  if (targetCalendarId) {
    const baseDesc = `会社名: ${company}
担当: ${sheet.getRange(row, CONFIG.A_STAFF).getValue() || ''}
商談担当: ${rep}
資料後追い結果: ${result}
6/27イベント: ${june || '未設定'}
8/29イベント: ${aug || '未設定'}
11月イベント: ${nov || '未設定'}

最新メモ:
${memo || ''}`;
    const nextId = upsertReminderCalendarEvent_(targetCalendarId, sheet.getRange(row, CONFIG.A_NEXT_CAL_ID).getValue(), company, '資料後追い', nextDate, nextMethod, memo || result, baseDesc);
    if (nextId !== null) sheet.getRange(row, CONFIG.A_NEXT_CAL_ID).setValue(nextId || '');
  }

  clearTargetCheckbox_(sheet, row, CONFIG.A_TARGET);
  const msg = result === 'アポ確定'
    ? '資料送付後追いを記録しました。アポ確定に更新したので、続けて「アポセットアップ」を押してください。'
    : '資料送付後追いを記録しました。次回後追い日を入れた場合は info カレンダーにも反映しています。';
  SpreadsheetApp.getActiveSpreadsheet().toast(msg, '📞 資料送付後追い', 5);
  return msg;
  });
}

function submitMeetingOutcomeRecord(payload) {
  if (!ensurePermissions()) throw new Error('権限承認が必要です。');
  ensureActiveSheetStructure_();
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(CONFIG.SS_ACTIVE);
  ensureActiveSheetStructure_();
  const row = resolveActiveRow_(sheet, payload);
  if (!sheet || row <= 1) throw new Error('アクティブ案件の対象行が見つかりません。');
  const settings = getSettings();
  const targetCalendarId = settings.infoCalendarId || '';
  const company = sheet.getRange(row, CONFIG.A_COMPANY).getValue() || '';
  const staff = sheet.getRange(row, CONFIG.A_STAFF).getValue() || '';
  const meetingRep = sheet.getRange(row, CONFIG.A_MEETING_REP).getValue() || sheet.getRange(row, CONFIG.A_APPOINTER).getValue() || '';
  const mode = String(payload.mode || '通常更新');
  const june = String(payload.june || '');
  const aug = String(payload.aug || '');
  const nov = String(payload.nov || '');
  const nextMonth = String(payload.nextMonth || '');
  const nextDate = parseFlexibleDateTime(payload.nextDate);
  const nextMethod = String(payload.nextMethod || '');
  const nextContent = String(payload.nextContent || '').trim();
  const midDate = parseFlexibleDateTime(payload.midDate);
  const midMethod = String(payload.midMethod || '');
  const midContent = String(payload.midContent || '').trim();
  const finalDate = parseFlexibleDateTime(payload.finalDate);
  const finalMethod = String(payload.finalMethod || '');
  const finalContent = String(payload.finalContent || '').trim();
  const memo = String(payload.memo || '').trim();

  if (payload.nextDate && !nextDate) throw new Error('次回アプローチ日の形式が正しくありません。');
  if (payload.midDate && !midDate) throw new Error('中間アプローチ日の形式が正しくありません。');
  if (payload.finalDate && !finalDate) throw new Error('最終決定アプローチ日の形式が正しくありません。');
  const guardKey = `MTG:${payload.recordId || company}:${mode}:${payload.nextDate}:${payload.midDate}:${payload.finalDate}:${memo}`;

  syncToFirestore_(meetingRep, `商談済:${mode}`, '商談完了');
  return withSubmissionGuard_(guardKey, 15, () => {

  sheet.getRange(row, CONFIG.A_JUNE_STATUS).setValue(june);
  sheet.getRange(row, CONFIG.A_AUG_STATUS).setValue(aug);
  sheet.getRange(row, CONFIG.A_NOV_STATUS).setValue(nov);
  sheet.getRange(row, CONFIG.A_NEXT_MONTH).setValue(nextMonth);
  if (nextDate) sheet.getRange(row, CONFIG.A_NEXT_APPROACH_DATE).setValue(nextDate);
  if (nextMethod) sheet.getRange(row, CONFIG.A_NEXT_METHOD).setValue(nextMethod);
  if (nextContent || payload.nextDate || nextMethod) sheet.getRange(row, CONFIG.A_NEXT_CONTENT).setValue(nextContent);
  if (midDate) sheet.getRange(row, CONFIG.A_MID_DATE).setValue(midDate);
  if (midMethod || payload.midDate) sheet.getRange(row, CONFIG.A_MID_METHOD).setValue(midMethod);
  if (midContent || payload.midDate || midMethod) sheet.getRange(row, CONFIG.A_MID_CONTENT).setValue(midContent);
  if (finalDate) sheet.getRange(row, CONFIG.A_FINAL_DATE).setValue(finalDate);
  if (finalMethod || payload.finalDate) sheet.getRange(row, CONFIG.A_FINAL_METHOD).setValue(finalMethod);
  if (finalContent || payload.finalDate || finalMethod) sheet.getRange(row, CONFIG.A_FINAL_CONTENT).setValue(finalContent);

  const actionCount = Number(sheet.getRange(row, CONFIG.A_ACTION_COUNT).getValue()) || 0;
  const today = new Date();
  const todayStr = Utilities.formatDate(today, 'Asia/Tokyo', 'M/d');
  const prefix = ['①','②','③','④','⑤','⑥','⑦','⑧','⑨','⑩'][actionCount] || `(${actionCount + 1})`;
  const resultText = `${mode} / 6月:${june || '未設定'} 8月:${aug || '未設定'} 11月:${nov || '未設定'}`;
  const extraNotes = [memo, nextContent, midContent, finalContent].filter(Boolean).join(' / ');
  const historyEntry = `${prefix}${todayStr} ${meetingRep || '担当未設定'}:${resultText}${extraNotes ? ' [' + extraNotes + ']' : ''}`;
  appendMultilineCell_(sheet.getRange(row, CONFIG.A_HISTORY), historyEntry);
  if (memo) appendMultilineCell_(sheet.getRange(row, CONFIG.A_MEET_MEMO), `[${todayStr}] ${memo}`);
  sheet.getRange(row, CONFIG.A_ACTION_COUNT).setValue(actionCount + 1);
  sheet.getRange(row, CONFIG.A_LAST_ACTION).setValue(today);
  sheet.getRange(row, CONFIG.A_LAST_RESULT).setValue(mode);
  sheet.getRange(row, CONFIG.A_MEET_DONE).setValue(true);
  sheet.getRange(row, CONFIG.A_KIND).setValue('商談済');

  if (mode === '出展決定' || [june, aug, nov].some(v => v === '出展決定')) {
    won = true;
    sheet.getRange(row, CONFIG.A_WON).setValue(true);
    sheet.getRange(row, CONFIG.A_KIND).setValue('クローズ');
  }
  if (mode === '失注' || ([june, aug, nov].every(v => v === '見送り') && [june, aug, nov].some(Boolean))) {
    sheet.getRange(row, CONFIG.A_LOST_DATE).setValue(today);
    if (!sheet.getRange(row, CONFIG.A_LOST_REASON).getValue()) sheet.getRange(row, CONFIG.A_LOST_REASON).setValue('ニーズ不一致');

    const baseDesc = `会社名: ${company}\n担当者: ${staff}\n商談担当: ${meetingRep}\n6/27イベント: ${june || '未設定'}\n8/29イベント: ${aug || '未設定'}\n11月イベント: ${nov || '未設定'}\n\n最新メモ:\n${memo || ''}`;

  return msg;
  });
}

function upsertReminderCalendarEvent_(calendarId, existingEventId, company, label, dateObj, method, content, baseDesc) {
  if (!dateObj) return null;
  const start = new Date(dateObj.getTime());
  if (start.getHours() === 0 && start.getMinutes() === 0) {
    start.setHours(9, 0, 0, 0);
  }
  const end = new Date(start.getTime() + 30 * 60 * 1000);
  const summary = `【${company}】${label}`;
  const description = `${baseDesc}\n\nアプローチ種別: ${label}\n方法: ${method || '未設定'}\n内容: ${content || '未設定'}`;
  const resource = {
    summary: summary,
    description: description,
    start: { dateTime: start.toISOString(), timeZone: 'Asia/Tokyo' },
    end: { dateTime: end.toISOString(), timeZone: 'Asia/Tokyo' },
  };
  try {
    if (existingEventId) {
      const updated = Calendar.Events.patch(resource, calendarId, existingEventId, { sendUpdates: 'none' });
      return updated.id;
    }
  } catch (e) {
    // 既存IDが消えていたら新規作成にフォールバック
  }
  const created = Calendar.Events.insert(resource, calendarId, { sendUpdates: 'none' });
  return created.id;
}

function appendMultilineCell_(range, text) {
  const value = String(text || '').trim();
  if (!value) return;
  const existing = String(range.getValue() || '').trim();
  range.setValue(existing ? existing + '\n' + value : value);
}

function escapeHtml(value) {
  return String(value || '')
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

function formatDateForDatetimeLocal_(value) {
  if (!value) return '';
  const d = new Date(value);
  if (isNaN(d.getTime())) return '';
  return Utilities.formatDate(d, 'Asia/Tokyo', 'yyyy-MM-dd\'T\'HH:mm');
}

function formatDateForTextInput_(value) {
  if (!value) return '';
  const d = new Date(value);
  if (isNaN(d.getTime())) return String(value || '');
  if (d.getHours() === 0 && d.getMinutes() === 0) {
    return Utilities.formatDate(d, 'Asia/Tokyo', 'yyyy/MM/dd');
  }
  return Utilities.formatDate(d, 'Asia/Tokyo', 'yyyy/MM/dd HH:mm');
}

function parseFlexibleDateTime(input, forceDateTimeLocal) {
  if (input === null || input === undefined) return null;
  const raw = String(input).trim();
  if (!raw) return null;

  if (forceDateTimeLocal) {
    const direct = new Date(raw);
    if (!isNaN(direct.getTime())) return direct;
  }

  const normalized = raw.replace(/-/g, '/').replace('T', ' ').trim();
  let m = normalized.match(/^(\d{4})\/(\d{1,2})\/(\d{1,2})(?:\s+(\d{1,2}):(\d{2}))?$/);
  if (m) {
    const d = new Date(Number(m[1]), Number(m[2]) - 1, Number(m[3]), Number(m[4] || 0), Number(m[5] || 0), 0, 0);
    return isNaN(d.getTime()) ? null : d;
  }
  m = normalized.match(/^(\d{1,2})\/(\d{1,2})(?:\s+(\d{1,2}):(\d{2}))?$/);
  if (m) {
    const now = new Date();
    const d = new Date(now.getFullYear(), Number(m[1]) - 1, Number(m[2]), Number(m[3] || 0), Number(m[4] || 0), 0, 0);
    if (d < new Date(now.getFullYear(), now.getMonth(), now.getDate()) && !m[3]) d.setFullYear(d.getFullYear() + 1);
    return isNaN(d.getTime()) ? null : d;
  }
  const direct = new Date(raw);
  return isNaN(direct.getTime()) ? null : direct;
}

function columnNumberToLetter(n) {
  let s = '';
  while (n > 0) {
    const m = (n - 1) % 26;
    s = String.fromCharCode(65 + m) + s;
    n = Math.floor((n - 1) / 26);
  }
  return s;
}
/** データをアプリ(Firestore)に送信するヘルパー */
function syncToFirestore_(memberId, type, industry, count = 1) {
  try {
    const timestamp = new Date();
    const dateStr = Utilities.formatDate(timestamp, "Asia/Tokyo", "yyyy-MM-dd");
    const payload = {
      fields: {
        memberId: { stringValue: String(memberId) },
        type: { stringValue: String(type) },
        industry: { stringValue: String(industry || '未設定') },
        timestamp: { timestampValue: timestamp.toISOString() },
        count: { integerValue: count },
        date: { stringValue: dateStr }
      }
    };
    const url = `https://firestore.googleapis.com/v1/projects/${CONFIG.FIREBASE_PROJECT_ID}/databases/(default)/documents/kpi_sync?key=${CONFIG.FIREBASE_API_KEY}`;
    UrlFetchApp.fetch(url, {
      method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true
    });
  } catch (e) {
    console.error("Firestore Sync Error: " + e.message);
  }
}

/** スプシのメンバー情報をアプリに一括同期する */
function syncMembersToFirestore() {
  const settings = getSettings();
  for (const name in settings.members) {
    const m = settings.members[name];
    const payload = {
      fields: {
        name: { stringValue: name },
        email: { stringValue: m.email || "" },
        full: { stringValue: m.full || name },
        id: { stringValue: name }
      }
    };
    const url = `https://firestore.googleapis.com/v1/projects/${CONFIG.FIREBASE_PROJECT_ID}/databases/(default)/documents/members/${encodeURIComponent(name)}?key=${CONFIG.FIREBASE_API_KEY}`;
    try {
      UrlFetchApp.fetch(url, { method: "patch", contentType: "application/json", payload: JSON.stringify(payload) });
    } catch(e) {
      UrlFetchApp.fetch(`https://firestore.googleapis.com/v1/projects/${CONFIG.FIREBASE_PROJECT_ID}/databases/(default)/documents/members?documentId=${encodeURIComponent(name)}&key=${CONFIG.FIREBASE_API_KEY}`, {
        method: "post", contentType: "application/json", payload: JSON.stringify(payload)
      });
    }
  }
  SpreadsheetApp.getUi().alert("✅ メンバー情報をアプリに同期しました。");
}

/**
 * 過去データ一括同期用スクリプト（この関数を1回だけ手動で実行します）
 */
function syncHistoricalData() {
  // ① 対象のシート名に変更してください（例: "アクティブ案件" や "企業マスター" など）
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("企業マスター"); 
  
  if (!sheet) {
    Logger.log("シートが見つかりません。シート名を確認してください。");
    return;
  }

  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  if (lastRow < 2) return; 
  
  const data = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues();

  // ② 列の番号を設定してください（配列は0から始まるため A列=0, B列=1, C列=2... となります）
  // ★ ここはご自身のスプレッドシートの列に合わせて必ず数字を書き換えてください！
  const DATE_COL = 11;       // L列
  const INDUSTRY_COL = 7;   // H列
  const APPOINTER_COL = 3;  // D列
  const STATUS_COL = 1;     // B列
  
  // プロジェクトID（現在のコードの上部で定義されている FIREBASE_PROJECT_ID を指定します）
  // もしエラーになる場合は "ivent-kpi" などの実際のプロジェクトIDを直接 "" の中に入れてください。
    const PROJECT_ID = "ivent-kpi";

  
  let successCount = 0;

  for (let i = 0; i < data.length; i++) {
    const row = data[i];
    const recordDate = row[DATE_COL]; 
    const industry = row[INDUSTRY_COL];
    const appointer = row[APPOINTER_COL];
    const status = row[STATUS_COL];
    
    // アポ獲得者かステータスが空欄の行はスキップ
    if (!appointer || !status) continue;
    
    try {
       // スプレッドシートの日付をそのまま保持する（空の場合は今日の日付にする）
       const timestamp = (recordDate instanceof Date) ? recordDate : new Date();
       const dateStr = Utilities.formatDate(timestamp, "Asia/Tokyo", "yyyy-MM-dd");
       
       const payload = {
         fields: {
           memberId: { stringValue: String(appointer) },
           type: { stringValue: String(status) }, 
           industry: { stringValue: String(industry || '未設定') },
           timestamp: { timestampValue: timestamp.toISOString() }, // ←ここが過去の日付になるのが重要です
           count: { integerValue: 1 },
           date: { stringValue: dateStr }
         }
       };

       // kpi_syncコレクションへ送信
       const url = `https://firestore.googleapis.com/v1/projects/${PROJECT_ID}/databases/(default)/documents/kpi_sync`;
       
       const options = {
         method: "post",
         contentType: "application/json",
         payload: JSON.stringify(payload),
         muteHttpExceptions: true
       };

       UrlFetchApp.fetch(url, options);
       successCount++;
       
       // 大量データによるGoogleのAPI制限エラーを回避するため、50件ごとに1秒ストップさせる
       if (successCount % 50 === 0) {
         Utilities.sleep(1000); 
       }
       
    } catch (e) {
      Logger.log("エラー行 " + (i + 2) + ": " + e.message);
    }
  }
  
  Logger.log("同期完了！ " + successCount + "件の過去データをFirebaseへ移行しました。");
}
